\subsubsection{Metodo di somiglianza: $f(x)$ polinomio}
Utile per trovare la soluzione particolare $y_p$ delle EDO di II ordine non omogenee.
\[
	ay''+by'+cy=f(x)
\]
\textcolor{red}{$f(x)$ ├¿ un polinomio di grado $n$}:
\begin{esercizio}[Caso 1: $c \neq 0$]
	\textcolor{red}{$c \neq 0$}
	\[
		y''+2y'-8y=x^2+3x+1
	\]
	Qui $f(x)=x^2+3x+1$ che ├¿ un polinomio con
	\begin{itemize}
		\item $a=1$
		\item $b=2$
		\item $c=-8$
	\end{itemize}
	$c \neq 0$ quindi devo cercare polinomio di grado $n$ (che corrisponde al grado del polinomio a destra dell'uguale), quindi $n=2$:
	\begin{enumerate}
		\item Scrivo polinomio di grado $n=2$ generico:
		      \[
			      \alpha x^2+\beta x + \gamma
		      \]
		      Questa sar├á la forma della soluzione particolare $y_p$
		\item Devo trovare chi sono $\alpha,\beta,\gamma$, come faccio?
		      Calcolo derivata prima e seconda della soluzione particolare: $y_p=\alpha x^2+\beta x + \gamma$ \\
		      \begin{itemize}
			      \item Calcolo derivata prima:
			            \[
				            y'_p=2 \alpha x+\beta
			            \]
			      \item Calcolo derivata seconda:
			            \[
				            y''_p=2 \alpha
			            \]
		      \end{itemize}
		\item Sostituisco $y_p,y'_p, y''_p$ trovati in $y''+2y'-8y$: \\
		      Ho:
		      \[
			      y''+2y'-8y=x^2+3x+1
		      \]
		      Sostituisco $y_p=y,y'_p=y', y''_p=y''$
		      \[
			      (2 \alpha)+2(2 \alpha x+\beta)-8(\alpha x^2+\beta x + \gamma)=x^2+3x+1
		      \]
		      Ora sviluppo i calcoli per trovare $\alpha,\beta,\gamma$:
		      \begin{align*}
			      (2 \alpha)+2(2 \alpha x+\beta)-8(\alpha x^2+\beta x + \gamma)      & = x^2+3x+1 \\
			      2 \alpha+4 \alpha x+2\beta-8 \alpha x^2 -8 \beta x -8 \gamma       & = x^2+3x+1 \\
			      x^2(-8 \alpha)+x(4 \alpha -8 \beta) + 2 \alpha + 2\beta - 8 \gamma & = x^2+3x+1
		      \end{align*}
		      Metto a sistema: \\
		      \textcolor{red}{$x^2(-8 \alpha)$}$+$\textcolor{blue}{$x(4 \alpha -8 \beta)$} $+$ \textcolor{orange}{$2 \alpha + 2\beta - 8 \gamma$} $=$ \textcolor{red}{$x^2$} $+$ \textcolor{blue}{$3x$} $+$ \textcolor{orange}{$1$}
		      \[
			      \begin{cases}
				      -8 \alpha = 1          \\
				      4 \alpha - 8 \beta = 3 \\
				      2 \alpha + 2 \beta - 8 \gamma = 1
			      \end{cases}
		      \]
		      Risolvo il sistema e ottengo:
		      \[
			      \begin{cases}
				      \alpha = - \frac{1}{8} \\
				      \beta = - \frac{7}{16} \\
				      \gamma = - \frac{17}{64}
			      \end{cases}
		      \]
		\item La soluzione particolare $y_p=\alpha x^2+\beta x + \gamma$ dopo aver trovato $\alpha, \beta, \gamma$ sar├á:
		      \begin{align*}
			      y_p
			       & = \alpha x^2+\beta x + \gamma                        \\
			       & = - \frac{1}{8} x^2 - \frac{7}{16} x - \frac{17}{64}
		      \end{align*}
	\end{enumerate}
\end{esercizio}

\begin{esercizio}[Caso 2: $c$ uguale a zero]
	\textcolor{red}{$c = 0, \quad b \neq 0$}
	\[
		y''+7y'=x+3
	\]
	$c=0$ quindi devo cercare polinomio di grado $n+1$, poich├¿ il grado di $x+3$ ├¿ 1, allora $n+1=1+1=2$, quindi $n=2$:
	\begin{enumerate}
		\item Scrivo polinomio di grado $n=2$ generico:
		      \[
			      y_p=\alpha x^2+\beta x + \gamma
		      \]
		\item Calcolo derivata prima e seconda
		      \begin{itemize}
			      \item $y'_p=2\alpha x+\beta$
			      \item $y''_p=2\alpha$
		      \end{itemize}
		\item Sostituisco nella EDO originale e trovo $\alpha, \beta, \gamma$: \\
		      Sostituisco $y_p,y'_p, y''_p$ trovati in $y''+7y'$:
		      \[
			      y''+7y'=x+3
		      \]
		      \[
			      (2 \alpha)+7(2\alpha x+\beta)=x+3
		      \]
		      \[
			      2 \alpha+14\alpha x+7\beta=x+3
		      \]
		      \[
			      \textcolor{red}{x(14\alpha)}+\textcolor{blue}{2\alpha+7\beta}=\textcolor{red}{x}+\textcolor{blue}{3}
		      \]
		      Metto a sistema:
		      \[
			      \begin{cases}
				      14 \alpha = 1 \\
				      2 \alpha + 7 \beta =3
			      \end{cases}
		      \]
		      Risolvo il sistema e ottengo:
		      \[
			      \begin{cases}
				      \alpha = \frac{1}{14} \\
				      \beta = \frac{20}{49}
			      \end{cases}
		      \]
		\item Ottengo la soluzione particolare: \\
		      La soluzione particolare $y_p=\alpha x^2+\beta x + \gamma$ dopo aver trovato $\alpha, \beta, \gamma$ sar├á:
		      \begin{align*}
			      y_p
			       & = \alpha x^2+\beta x + \gamma            \\
			       & = \frac{1}{14} x^2 + \frac{20}{49} x - 0 \\
			       & = \frac{1}{14} x^2 + \frac{20}{49} x
		      \end{align*}
	\end{enumerate}
\end{esercizio}

\begin{esercizio}[Caso 3]
	\textcolor{red}{$c,b = 0$}
	\[
		y''=x^2+4x-7
	\]
	$c,b=0$ quindi devo cercare un polinomio di grado $n+2$, poich├¿ il grado di $x^2+4x-7$ ├¿ 2, allora $n+2=2+2=4$. \\ \\
	\textcolor{red}{Ho due metodi}:
	\begin{itemize}
		\item Integrare due volte, oppure
		\item Trovare il polinomio generico di grado $n=4$:
		      \[
			      \alpha x^4+\beta x^3+\gamma x^2+\delta x + \epsilon
		      \]
	\end{itemize}
	\textcolor{blue}{\textbf{Primo metodo: Integrare due volte}} \\
	Qui non trovo la soluzione particolare ma trovo la soluzione generale della EDO:
	\begin{align*}
		\int(x^2+4x-7)dx
		 & = \int x^2dx + \int 4x dx - \int 7 dx \\
		 & = \frac{x^3}{3} + 2x^2 - 7x + c_1
	\end{align*}
	Integro di nuovo:
	\begin{align*}
		\int\left(\frac{x^3}{3} + 2x^2 - 7x + c_1\right)dx
		 & = \int \frac{x^3}{3}dx + \int 2x^2 dx - \int 7x dx + \int c_1 dx \\
		 & = \frac{x^4}{12} + \frac{2x^3}{3} - \frac{7x^2}{2} + c_1x + c_2
	\end{align*}
	Soluzione generale della EDO:
	\[
		\frac{x^4}{12} + \frac{2x^3}{3} - \frac{7x^2}{2} + c_1x + c_2
	\]
	\textcolor{blue}{\textbf{Secondo metodo: metodo di somiglianza}} \\
	Qui trovo solo la soluzione particolare:
	\begin{enumerate}
		\item Scrivo polinomio di grado $n=4$ generico:
		      \[
			      y_p=\alpha x^4+\beta x^3+\gamma x^2+\delta x + \epsilon
		      \]
		\item Calcolo derivata seconda del polinomio generico:
		      \begin{itemize}
			      \item $y'_p=4\alpha x^3+3\beta x^2+2 \gamma x + \delta$
			      \item $y''_p=12\alpha x^2+6\beta x+2 \gamma$
		      \end{itemize}
		\item Sostituisco la derivata seconda trovata nella EDO e trovo $\alpha, \beta, \gamma, \delta, \epsilon$:
		      \[
			      y''=x^2+4x-7
		      \]
		      diventa:
		      \[
			      (12\alpha x^2+6\beta x+2 \gamma)=x^2+4x-7
		      \]
		      \[
			      x^2(12 \alpha)+x(6 \beta)+2 \gamma=x^2+4x-7
		      \]
		      Metto a sistema:
		      \[
			      \begin{cases}
				      12 \alpha = 1 \\
				      6 \beta =4    \\
				      2 \gamma =-7  \\
			      \end{cases}
		      \]
		      Risolvo il sistema e ottengo:
		      \[
			      \begin{cases}
				      \alpha = \frac{1}{12} \\
				      \beta = \frac{2}{3}   \\
				      \gamma = - \frac{7}{2}
			      \end{cases}
		      \]
		\item Ottengo la soluzione particolare:
		      La soluzione particolare $y_p=\alpha x^4+\beta x^3+\gamma x^2+\delta x + \epsilon$ dopo aver trovato $\alpha, \beta, \gamma, \delta, \epsilon$ sar├á:
		      \begin{align*}
			      y_p
			       & = \alpha x^4+\beta x^3+\gamma x^2+\delta x + \epsilon           \\
			       & = \frac{1}{12} x^4 + \frac{2}{3} x^3 - \frac{7}{2} x^2 + 0x + 0 \\
			       & = \frac{1}{12} x^4 + \frac{2}{3} x^3 - \frac{7}{2} x^2
		      \end{align*}
	\end{enumerate}
\end{esercizio}

\subsubsection{Metodo di somiglianza: $f(x)$ esponenziale}
Utile per trovare la soluzione particolare $y_p$ delle EDO di II ordine non omogenee.
\[
	ay''+by'+cy=f(x)
\]
\textcolor{red}{$f(x)$ ├¿ un esponenziale} \\
\begin{esercizio}[Caso 1: $\alpha$ non ├¿ soluzione dell'eq. caratteristica]
	\textcolor{red}{$f(x)=c \cdot e^{\alpha x}$ e $\alpha$ \textcolor{black}{non} ├¿ soluzione dell'equazione caratteristica.}
	\[
		y''+7y'+12y=5e^{2x}
	\]
	In questo caso abbiamo che $c=5$ e $\alpha=2$. \\
	\begin{enumerate}
		\item Trovo soluzioni dell'omogenea associata (ponendo $f(x)=0$):
		      \[
			      y''+7y'+12y=0
		      \]
		      Quindi:
		      \[
			      \lambda^2 + 7 \lambda + 12 = 0
		      \]
		      \[
			      \Delta=b^2-4ac=49-4(12)=49-48=1
		      \]
		      \[
			      x_1,x_2=\frac{-b \pm \sqrt{\Delta}}{2a}=\frac{-7 \pm 1}{2}=
		      \]
		      \\
		      \[
			      x_1=-3
		      \]
		      \[
			      x_2=-4
		      \]
		\item $\alpha$ fa parte delle soluzioni dell'equazione caratteristica? \\
		      No, perch├¿ ho ottenuto $x_1=-3,x_2=-4$. Quindi adesso cerco soluzione particolare nella forma:
		      \[
			      y_p=k\cdot e^{\alpha x}
		      \]
		      \begin{enumerate}
			      \item Calcolo derivata prima e seconda di $y_p=k\cdot e^{2x}$:
			            \begin{itemize}
				            \item $y'_p=2ke^{2x}$
				            \item $y''_p=4ke^{2x}$
			            \end{itemize}
			      \item Sostituisco le derivate nella EDO:
			            \[
				            y''+7y'+12y=5e^{2x}
			            \]
			            diventa:
			            \[
				            (4ke^{2x})+7(2ke^{2x})+12(k\cdot e^{2x})=5e^{2x}
			            \]
			            \[
				            ke^{2x}\left(4+ (7 \cdot 2) + 12\right)=5e^{2x}
			            \]
			            \[
				            30ke^{2x}=5e^{2x}
			            \]
			            Ricavo $k$:
			            \[
				            k=\frac{5e^{2x}}{30e^{2x}}=\frac{5}{30}\cdot \frac{e^{2x}}{e^{2x}}=\frac{1}{6} \cdot (e^{2x-2x})= \frac{1}{6} \cdot e^0= \frac{1}{6}
			            \]
			      \item La soluzione particolare $y_p=k\cdot e^{\alpha x}$ diventa:
			            \[
				            y_p=\frac{1}{6}e^{2x}
			            \]
		      \end{enumerate}
	\end{enumerate}
	Quindi la soluzione della EDO sar├á $y=y_O+y_p$, ovvero:
	\[
		y=\left(c_1 e^{-3x}+c_2 e^{-4x}\right)+ \frac{1}{6}e^{2x}
	\]
\end{esercizio}

\begin{esercizio}[Caso 2: $\alpha$ ├¿ soluzione dell'eq. caratteristica]
	\textcolor{red}{$f(x)=c \cdot e^{\alpha x}$ e $\alpha$ ├¿ soluzione dell'equazione caratteristica.}
	\[
		3y''-20y'-7y=4e^{7x}
	\]
	In questo caso abbiamo che $c=4$ e $\alpha=7$. \\
	\begin{enumerate}
		\item Trovo soluzioni dell'omogenea associata (ponendo $f(x)=0$):
		      \[
			      3y''-20y'-7y=0
		      \]
		      Quindi:
		      \[
			      3\lambda^2 - 20 \lambda -7 = 0
		      \]
		      \[
			      \Delta=b^2-4ac=400-4(3)(-7)=400+84=484
		      \]
		      \[
			      x_1,x_2=\frac{-b \pm \sqrt{\Delta}}{2a}=\frac{20 \pm \sqrt{484}}{6}=\frac{-20 \pm 22}{6}=
		      \]
		      \\
		      \[
			      x_1=-\frac{1}{3}
		      \]
		      \[
			      x_2=7
		      \]
		\item $\alpha$ fa parte delle soluzioni dell'equazione caratteristica? \\
		      S├¼, perch├¿ ho ottenuto $x_1=-\frac{1}{3},x_2=7$. Quindi adesso cerco soluzione particolare nella forma:
		      \[
			      y_p=kx \cdot e^{\alpha x}
		      \]
		      \begin{enumerate}
			      \item Calcolo derivata prima e seconda di $y_p=kx \cdot e^{7x}$:
			            \begin{itemize}
				            \item \begin{align*}
					                  y'_p
					                   & = \bigl( f' \cdot g + f \cdot g' \bigr) \\[4pt]
					                   & = \bigl( (k x)' \cdot e^{7x}
					                  + k x \cdot (e^{7x})' \bigr)               \\[4pt]
					                   & = k e^{7x} + 7k x e^{7x}
				                  \end{align*}

				            \item \begin{align*}
					                  y''_p
					                   & = \textcolor{red}{(k e^{7x})'} + (7k x e^{7x})'     \\[4pt]
					                   & = \textcolor{red}{7k e^{7x}}
					                  + \bigl( (7k x)' \cdot e^{7x}
					                  + 7k x \cdot (e^{7x})' \bigr)                          \\[4pt]
					                   & = 7k e^{7x} + \left(7k e^{7x} + 49k x e^{7x}\right) \\[4pt]
					                   & = 14k e^{7x}+49k x e^{7x}
				                  \end{align*}
			            \end{itemize}
			      \item Sostituisco le derivate nella EDO e trovo $k$:
			            \[
				            3y''-20y'-7y=4e^{7x}
			            \]
			            diventa:
			            \[
				            3(14k e^{7x}+49k x e^{7x})-20(k e^{7x} + 7k x e^{7x} )-7(kx \cdot e^{7x})=4e^{7x}
			            \]
			            \[
				            42k e^{7x}+147k x e^{7x}-20k e^{7x}-140k x e^{7x}-7kxe^{7x}=4e^{7x}
			            \]
			            \[
				            ke^{7x}(42+147x-20-140x-7x)=4e^{7x}
			            \]
			            \[
				            ke^{7x}(22)=4e^{7x}
			            \]
			            Ricavo $k$:
			            \[
				            22ke^{7x}=4e^{7x}
			            \]
			            \[
				            k=\frac{4e^{7x}}{22e^{7x}}=\frac{4}{22}=\frac{2}{11}
			            \]
			      \item La soluzione particolare $y_p=kx\cdot e^{\alpha x}$ diventa:
			            \[
				            y_p=\frac{2}{11}xe^{7x}
			            \]
		      \end{enumerate}
	\end{enumerate}
	Quindi la soluzione della EDO sar├á $y=y_O+y_p$, ovvero:
	\[
		y=\left(c_1 e^{-\frac{1}{3}x}+c_2 e^{7x}\right)+ \frac{2}{11}xe^{7x}
	\]
\end{esercizio}

\begin{esercizio}[Caso 3]
	\textcolor{red}{$f(x)=c \cdot e^{\alpha x}$ e \\ $\alpha= \lambda_1= \lambda_2$ ├¿ uguale ad entrambe le soluzioni dell'equazione caratteristica.}
	\[
		y''-12y'+36y=3e^{6x}
	\]
	In questo caso abbiamo che $c=3$ e $\alpha=6$. \\
	\begin{enumerate}
		\item Trovo soluzioni dell'omogenea associata (ponendo $f(x)=0$):
		      \[
			      y''-12y'+36y=0
		      \]
		      Quindi:
		      \[
			      \lambda^2-12\lambda+36=0
		      \]
		      \[
			      \Delta=b^2-4ac=144-144=0
		      \]
		      \[
			      \lambda_1,\lambda_2=\frac{-b \pm 0}{2a}=\frac{12}{2}=6
		      \]
		      Quindi ho due soluzioni coincidenti.
		\item Quando le soluzioni dell'equazione caratteristica coincidono e sono uguali ad $\alpha$, allora cerco soluzione particolare nella forma:
		      \[
			      y_p=kx^2 \cdot e^{\alpha x}
		      \]
		      \begin{enumerate}
			      \item Calcolo derivata prima e seconda di $y_p=kx^2 \cdot e^{6 x}$:
			            \begin{itemize}
				            \item \begin{align*}
					                  y'_p
					                   & = \bigl( f' \cdot g + f \cdot g' \bigr)                \\[4pt]
					                   & = \bigl((kx^2)' \cdot e^{6 x} + kx^2 \cdot (e^{6 x})') \\[4pt]
					                   & = 2kxe^{6 x} + 6kx^2e^{6 x}
				                  \end{align*}
				            \item \begin{align*}
					                  y''_p
					                   & = \bigl( f' \cdot g + f \cdot g' \bigr)                                                                       \\[4pt]
					                   & = \bigl((2kx)' \cdot e^{6 x} + 2kx \cdot (e^{6 x})') + \bigl((6kx^2)' \cdot e^{6 x} + 6kx^2 \cdot (e^{6 x})') \\[4pt]
					                   & = \left(2ke^{6 x} + 12kxe^{6 x}\right)+ \left(12kxe^{6 x} + 36kx^2e^{6 x}\right)                              \\
					                   & = 36kx^2e^{6 x} + 24kxe^{6 x} + 2ke^{6 x}
				                  \end{align*}
			            \end{itemize}
			      \item Sostiuisco le derivate nella EDO iniziale e ricavo $k$:
			            \[
				            y''-12y'+36y=3e^{6x}
			            \]
			            diventa:
			            \[
				            (36kx^2e^{6 x} + 24kxe^{6 x} + 2ke^{6 x})-12(2kxe^{6 x} + 6kx^2e^{6 x})+36(kx^2 e^{6 x})=3e^{6x}
			            \]
			            \[
				            36kx^2e^{6 x} + 24kxe^{6 x} + 2ke^{6 x}-24kxe^{6 x}-72kx^2e^{6 x}+ 36kx^2 e^{6 x}=3e^{6x}
			            \]
			            \[
				            2ke^{6 x}=3e^{6x}
			            \]
			            \[

\section{Propriet├á Vettoriali}
\subsection{Vettori \& Propriet├á}
\begin{definition}{Vettore}
	Il vettore $\vec{x} \in \mathbb{R}^n$ ├¿ una n-upla:
	\[
		\vec{x}=\left(x_1,x_2,\cdots,x_n\right)
	\]
\end{definition}
\begin{definition}{Prodotto scalare}
SSiano $\vec{x},\vec{y} \in \mathbb{R}^n$ due vettori
\begin{align*}
	\langle \vec{x},\vec{y} \rangle = \vec{x} \cdot \vec{y} = (x_1, \dots, x_n) \cdot (y_1, \cdots, y_n) = \sum_{i=1}^{n} \left(x_i \cdot y_i \right)
\end{align*}
Il \textcolor{red}{prodotto scalare} ├¿ un'operazione binaria che associa ad ogni coppia di vettori un numero reale, dato dalla somma dei prodotti delle componenti dei due vettori. \\ \\
Esempio in $\mathbb{R}^2$:
\begin{align*}
	\langle \vec{x},\vec{y} \rangle = \vec{x} \cdot \vec{y} = (x_1, x_2) \cdot (y_1, y_2) = x_1 y_1 + x_2 y_2
\end{align*}
\end{definition}
\subsubsection{Norma di un vettore}
\begin{definition}{Norma di un vettore}
	SSia $\vec{x} \in \mathbb{R}^n$ un vettore. Indico la \textcolor{red}{norma/lunghezza} di un vettore come la radice quadrata del prodotto scalare del vettore $\vec{x}$ per se stesso
	o equivalentemente, la radice quadrata della somma dei suoi componenti al quadrato, il numero reale non negativo:
	\[
		\textcolor{red}{\|\vec{x}\|}=\sqrt{\langle \vec{x}, \vec{x} \rangle}=\sqrt{x_1^2+x_2^2+ \cdots + x_n^2}=\sqrt{\sum_{i=1}^{n}(x_i)^2}
	\]
\end{definition}
\subsubsection{Formula di Carnot}
\begin{theorem}{Formula di Carnot}
	SSiano $\vec{x},\vec{y} \in \mathbb{R}^n$ due vettori. La norma al quadrato della somma dei due vettori equivale alle loro norme al quadrato pi├╣ due volte il loro prodotto scalare
	\footnote{Un p├▓ come $(a+b)^2=a^2+b^2+2ab$}:
	\[
		\textcolor{red}{\left\|\vec{x}+\vec{y}\right\|^2}=\left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2\langle \vec{x}, \vec{y} \rangle
	\]
	\begin{proof}
		Sapendo che $\|\vec{x}\|=\sqrt{\langle \vec{x}, \vec{x} \rangle}$, allora:
		\begin{align*}
			\textcolor{red}{\left\|\vec{x}+\vec{y}\right\|^2}
			 & = \left(\sqrt{\langle \vec{x} + \vec{y}, \vec{x} + \vec{y} \rangle}\right)^2                                                                \\
			 & = {\langle \vec{x} + \vec{y}, \vec{x} + \vec{y} \rangle}                                                                                    \\
			 & \textcolor{cyan}{\text{sviluppo prodotto scalare } \langle \text{arg1} ,  \text{arg2} \rangle}                                              \\
			 & = \langle \vec{x}, \vec{x} \rangle + \langle \vec{x}, \vec{y} \rangle + \langle \vec{y}, \vec{x} \rangle + \langle \vec{y}, \vec{y} \rangle \\
			 & \textcolor{cyan}{\text{dato che } \langle x,y \rangle = \langle y,x \rangle \text{ (simmetria) allora,}}                                    \\
			 & = \|\vec{x}\|^2+ 2\langle \vec{x}, \vec{y} \rangle + \|\vec{y}\|^2                                                                          \\
			 & = \|\vec{x}\|^2 + \|\vec{y}\|^2 + 2\langle \vec{x}, \vec{y} \rangle
		\end{align*}
	\end{proof}
\end{theorem}
\subsubsection{Disuguaglianza di Cauchy-Schwarz}
\begin{theorem}{Disuguaglianza di Cauchy-Schwarz p.56}
	SSiano $\vec{x}, \vec{y} \in \mathbb{R}^n$ due vettori. Il valore assoluto del prodotto scalare ├¿ minore o uguale $\leq$ al prodotto delle norme dei due vettori:
	\[
		|\langle \vec{x}, \vec{y} \rangle| \leq \|\vec{x}\| \cdot \|\vec{y}\|
	\]
	\begin{proof}
		Vogliamo dimostrare che $|\langle \vec{x}, \vec{y} \rangle| \leq \|\vec{x}\| \cdot \|\vec{y}\|$ \\
		\fbox{\textcolor{red}{Caso 1: $x \lor y = 0$}} 
		Supponiamo $\vec{y}=\vec{0}$
		\begin{align*}
			|\langle \vec{x}, \vec{y} \rangle| &\leq \|\vec{x}\| \cdot \|\vec{y}\| \\
			|\langle \vec{x}, \vec{0} \rangle| &\leq \|\vec{x}\| \cdot \|\vec{0}\| \\
			|\vec{x} \cdot \vec{0}| &\leq \sqrt{\sum_{i=1}^{n}(x_i)^2} \cdot \sqrt{\sum_{i=1}^{n}(0)^2} \\
			|(x_1 \cdot 0)+ \dots + (x_n \cdot 0)| &\leq 0 \\
			0 &\leq 0 \\
		\end{align*}
		\fbox{\textcolor{red}{Caso 2: $x,y \neq 0$}}
		Supponiamo $\vec{y} \neq 0$ e considero $\forall{t \in \mathbb{R}}$, il vettore:
		\begin{align*}
			\vec{z}(t)=\vec{x}+t\vec{y}
		\end{align*}
		e per la positivit├á della norma ho che $\|\vec{z}(t)\|^2 \geq 0$\footnote{
		si scrive $\|\vec{z}(t)\|^2$ e non $\|\vec{z}(t)\|$ perch├¿ elevando al quadrato si elimina la radice (serve per la dimostrazione).
		} per ogni $t$, quindi:
		\begin{align*}
			\|\vec{z}(t)\|^2 = \|\vec{x}+t\vec{y}\|^2 \overset{\text{Carnot}}{=} \|\vec{x}\|^2 + t^2\|\vec{y}\|^2 + 2t \langle \vec{x}, \vec{y} \rangle \geq 0
		\end{align*}
		questa $\|\vec{x}\|^2 + \textcolor{cyan}{t^2}\|\vec{y}\|^2 + 2 \textcolor{cyan}{t} \langle \vec{x}, \vec{y} \rangle$ ├¿ una funzione quadratica in $t$ della forma $at^2+bt+c$ con:
		\begin{itemize}
			\item $a = \|\vec{y}\|^2 > 0$ (ipotesi $\vec{y} \neq 0$)
			\item $b = 2\langle \vec{x}, \vec{y} \rangle$
			\item $c = \|\vec{x}\|^2$
		\end{itemize}
		essendo questa una parabola sempre non negativa ($\geq 0$) con $a > 0$, il discriminante deve soddisfare: 
		\begin{align*}
			\Delta = b^2-4ac &\leq 0 \\
			&\textcolor{cyan}{\text{sostituendo i valori}} \\
			(2 \langle \vec{x}, \vec{y} \rangle)^2-4(\|\vec{y}\|^2)(\|\vec{x}\|^2) &\leq 0 \\
			4 \langle \vec{x}, \vec{y} \rangle^2-4\|\vec{y}\|^2\|\vec{x}\|^2 &\leq 0 \\
			4 \langle \vec{x}, \vec{y} \rangle^2 &\leq 4\|\vec{y}\|^2\|\vec{x}\|^2 \\
			\langle \vec{x}, \vec{y} \rangle^2 &\leq \|\vec{y}\|^2\|\vec{x}\|^2 \\
			&\textcolor{cyan}{\text{applico radice e ottengo}} \\
			\langle \vec{x}, \vec{y} \rangle &\leq \|\vec{y}\| \cdot \|\vec{x}\|
		\end{align*}
	\end{proof}
	\begin{proof}
		Vogliamo dimostrare che $|\langle \vec{x}, \vec{y} \rangle| = \|\vec{x}\| \cdot \|\vec{y}\|$ \\
		\fbox{\textcolor{red}{Caso 1: $x \lor y = 0$}} 
		Supponiamo $\vec{y}=\vec{0}$
		\begin{align*}
			|\langle \vec{x}, \vec{y} \rangle| &= \|\vec{x}\| \cdot \|\vec{y}\| \\
			|\langle \vec{x}, \vec{0} \rangle| &= \|\vec{x}\| \cdot \|\vec{0}\| \\
			0 = 0
		\end{align*}
		\fbox{\textcolor{red}{Caso 2: $x,y \neq 0$}} \\
		L'uguaglianza $|\langle \vec{x}, \vec{y} \rangle| = \|\vec{x}\| \cdot \|\vec{y}\|$ si verifica quando $\Delta = 0$ quindi considero $\vec{x}=\lambda \vec{y}$ 
		con $\lambda \in \mathbb{R}, \lambda \neq 0$: \\
		\begin{enumerate}
			\item Sviluppo lato sinistro dell'uguaglianza
			\begin{align*}
			|\langle \vec{x}, \vec{y} \rangle| &= \\
			\textcolor{cyan}{\text{sostituisco } \vec{x} = \lambda \vec{y}}\\
			&= |\langle \lambda \vec{y}, \vec{y} \rangle|\\
			&= |\lambda| \cdot |\langle \vec{y}, \vec{y} \rangle|\\
			\textcolor{cyan}{\text{dato che } \langle \vec{x}, \vec{x} \rangle = \|\vec{x}\|^2, \text{ allora}}\\
			&= |\lambda| \cdot \|\vec{y}\|^2
		\end{align*}
			\item Sviluppo lato destro dell'uguaglianza
						\begin{align*}
			\|\vec{x}\| \cdot \|\vec{y}\| &= \\
			\textcolor{cyan}{\text{sostituisco } \vec{x} = \lambda \vec{y}}\\
			&= \|\lambda \vec{y}\| \cdot \|\vec{y}\| \\
			&= |\lambda| \cdot \|\vec{y}\|^2
		\end{align*}
			\item Unisco i due lati ottenuti nel passo 1 e nel passo 2 dimostrando che ├¿ vera l'uguaglianza 
		\end{enumerate}
		\begin{align*}
			|\lambda| \cdot \|\vec{y}\|^2 &= |\lambda| \cdot \|\vec{y}\|^2
		\end{align*}
	\end{proof}
	Come ho ottenuto $\lambda\vec{y}$? \\
	Quando $\Delta = 0$ vuol dire che $\exists t \in \mathbb{R}$ per cui la funzione si annulla, quindi impongo $f(t) = 0$:
		\begin{align*}
			\vec{x}+t\vec{y} = 0 \\
			\vec{x} = -t\vec{y} \\
			\textcolor{cyan}{\text{impongo } -t = \lambda} \\
			\vec{x} = \lambda\vec{y}
		\end{align*}
\end{theorem}
\subsubsection{Disuguaglianza triangolare}
\begin{theorem}{Disuguaglianza triangolare p.58}
	SSiano $\vec{x}, \vec{y} \in \mathbb{R}^n$ due vettori. La norma della somma dei due vettori ├¿ minore o uguale $\leq$ alla somma delle loro norme:
	\[
		\|\vec{x}+\vec{y}\| \leq \|\vec{x}\| + \|\vec{y}\|
	\]
	e se:
	\[
		\|\vec{x}+\vec{y}\| = \|\vec{x}\| + \|\vec{y}\| \text{ vuol dire che } \vec{y}=0 \lor \vec{x}=\lambda \cdot \vec{y} \text{ con } \lambda \geq 0
	\]
	\begin{proof}
		Questo teorema usa la formula di Carnot e Chauchy-Schwartz per la dimostrazione. \\
		\begin{align*}
			 & \textcolor{cyan}{\text{considero}}                                                                                                                                                                                                                                                                                                                                                 \\
			 & \|\vec{x}+\vec{y}\|                                                                                                                                                                                                                                                                                                                                                                \\
			 & \textcolor{cyan}{\text{elevo al quadrato}}                                                                                                                                                                                                                                                                                                                                         \\
			 & \left(\|\vec{x}+\vec{y}\| \right)^2                                                                                                                                                                                                                                                                                                                                                \\
			 & \|\vec{x}+\vec{y}\|^2                                                                                                                                                                                                                                                                                                                                                              \\
			 & \textcolor{cyan}{\text{so che }\left\|\vec{x}+\vec{y}\right\|^2 \overset{\text{Carnot}}{=} \left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2\langle \vec{x}, \vec{y} \rangle}
			\textcolor{cyan}{\text{ quindi: }}                                                                                                                                                                                                                                                                                                                                                    \\
			 & \left\|\vec{x}+\vec{y}\right\|^2 \overset{\text{Carnot}}{=} \left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2\langle \vec{x}, \vec{y} \rangle                                                                                                                                                                                                                                    \\
			 & \textcolor{cyan}{\text{so che: } \left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2\langle \vec{x}, \vec{y} \rangle \leq \left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2 |\langle \vec{x}, \vec{y} \rangle|}                                                                                                                                                                  \\
			 & \textcolor{cyan}{\text{perch├¿ ovviamente } \forall{z} \in \mathbb{R} \text{ vale sempre che } z \leq |z| \text{, quindi:}}                                                                                                                                                                                                                                                         \\
			 & \left\|\vec{x}+\vec{y}\right\|^2 \overset{\text{Carnot}}{=} \left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2\langle \vec{x}, \vec{y} \rangle \leq \left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2 |\langle \vec{x}, \vec{y} \rangle|                                                                                                                                        \\
			 & \textcolor{cyan}{\text{so che per Cauchy-Schwarz vale che: } |\langle \vec{x}, \vec{y} \rangle| \leq \left\|\vec{x}\right\| \left\|\vec{y}\right\| \text{ quindi:}}                                                                                                                                                                                                                \\
			 & \left\|\vec{x}+\vec{y}\right\|^2 \overset{\text{Carnot}}{=} \left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2\langle \vec{x}, \vec{y} \rangle \leq \left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2 |\langle \vec{x}, \vec{y} \rangle| \overset{\text{Cauchy-Schwarz}}{\leq} \left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2 \|\vec{x}\|\|\vec{y}\|                       \\
			\\&\textcolor{cyan}{\text{dato che:}} \\
			 & \left\|\vec{x}+\vec{y}\right\|^2 \overset{\text{Carnot}}{=} \left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2\langle \vec{x}, \vec{y} \rangle \leq \left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2 |\langle \vec{x}, \vec{y} \rangle| \leq \underbrace{\left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2 \|\vec{x}\|\|\vec{y}\|}_{=\left(\|\vec{x}\|+\|\vec{y}\|\right)^2} \\
			 & \textcolor{cyan}{\text{quindi:}}                                                                                                                                                                                                                                                                                                                                                   \\
			 & \left\|\vec{x}+\vec{y}\right\|^2 \overset{\text{Carnot}}{=} \left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2\langle \vec{x}, \vec{y} \rangle \leq \left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2 |\langle \vec{x}, \vec{y} \rangle| \leq {\left(\|\vec{x}\|+\|\vec{y}\|\right)^2}                                                                                          \\
			 & \textcolor{red}{\left\|\vec{x}+\vec{y}\right\|^2} \overset{\text{Carnot}}{=} \left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2\langle \vec{x}, \vec{y} \rangle \leq \left\|\vec{x}\right\|^2+\left\|\vec{y}\right\|^2+2 |\langle \vec{x}, \vec{y} \rangle| \textcolor{red}{\leq {\left(\|\vec{x}\|+\|\vec{y}\|\right)^2}}                                                        \\
			 & \textcolor{cyan}{\text{prendo ora le parti in rosso: }}                                                                                                                                                                                                                                                                                                                            \\
			 & \left\|\vec{x}+\vec{y}\right\|^2 \leq \left(\|\vec{x}\|+\|\vec{y}\|\right)^2                                                                                                                                                                                                                                                                                                       \\
			 & \textcolor{cyan}{\text{imposto radici per rimuovere le potenze di 2: }}                                                                                                                                                                                                                                                                                                            \\
			 & \sqrt{\left\|\vec{x}+\vec{y}\right\|^2} \leq \sqrt{\left(\|\vec{x}\|+\|\vec{y}\|\right)^2}                                                                                                                                                                                                                                                                                         \\
			 & \left\|\vec{x}+\vec{y}\right\| \leq \left(\|\vec{x}\|+\|\vec{y}\|\right)                                                                                                                                                                                                                                                                                                           \\
			 & \left\|\vec{x}+\vec{y}\right\| \leq \|\vec{x}\|+\|\vec{y}\|
		\end{align*}
		\footnotetext{Vale sempre:
			\[
				x \leq |x|  \quad \forall{x} \in \mathbb{R}
			\]ovvio che qualunque sia il valore di $x$, il suo valore assoluto sar├á sempre uguale o maggiore. Per esempio: se $x=-3$, allora $-3<|-3|$ perch├¿ diventa $-3<3$ ed ├¿ vero. Oppure, se $x=3$, allora $3=|3|$ perch├¿ $3=3$ ed ├¿ comunque vero.}
	\end{proof}
\end{theorem}

\section{Elementi di Topologia in $\mathbb{R}^n$}

\subsubsection{Distanza Euclidea}
\begin{definition}{Distanza euclidea}
	SSiano $\vec{x},\vec{y} \in \mathbb{R}^n$ due vettori. La \textcolor{red}{distanza euclidea} tra $\vec{x}$ e $\vec{y}$ ├¿ il numero reale non negativo definito dalla norma della loro differenza:
	\[
		d(\vec{x},\vec{y})=\left\|\vec{x}-\vec{y}\right\|=\sqrt{\langle \vec{x}-\vec{y}, \vec{x}-\vec{y} \rangle}=\sqrt{\sum_{i=1}^{n}(x_i-y_i)^2}
	\]
	Propriet├á:
	\begin{enumerate}
		\item $d(\vec{x},\vec{y}) \geq 0, \quad d(\vec{x},\vec{y}) \in \mathbb{R}$
		\item $d(\vec{x},\vec{y}) = 0 \iff \vec{x}=\vec{y}$
		\item $d(\vec{x},\vec{y}) = d(\vec{y},\vec{x})$ \hfill (simmetria)
		\item $d(\vec{x},\vec{y}) \leq d(\vec{x},\vec{z}) + d(\vec{z},\vec{y})$ \hfill (disugualianza triangolare)\\ \\
		      La distanza diretta tra $\vec{x},\vec{y}$ ├¿ minore o uguale al percorso che passa per un punto intermedio $\vec{z}$.
	\end{enumerate}
\end{definition}
La distanza euclidea ├¿ semplicemente un numero reale positivo che indica quanto sono distanti due punti nello spazio.
Per esempio, siano due punti $A,B$ in $\mathbb{R}^3$ con le seguenti coordinate cartesiane:
\[
	A(x_A, y_A, z_A), \quad B(x_B, y_B, z_B)
\]
\begin{center}
	\includegraphics[width=0.5\linewidth]{dist-euclidea.png}
\end{center}
Allora la distanza tra i due punti $A,B$ ├¿ la seguente
\[
	d(A,B)=d(\vec{OA},\vec{OB})=\|\vec{OB}-\vec{OA}\|=\|\vec{AB}\|=\sqrt{(x_B-x_A)^2+(y_B-y_A)^2+(z_B-z_A)^2}
\]


\subsubsection{Spazio Metrico}
\begin{definition}{Spazio Metrico}
	SSia $X$ un insieme qualsiasi. $X$ viene definito \textcolor{red}{spazio metrico} quando in esso viene introdotta una funzione distanza. \\ \\
	Uno spazio metrico ├¿ una coppia $\left(X, d\right)$ dove:
	\begin{itemize}
		\item $X$ ├¿ un insieme
		\item $d$ ├¿ una funzione distanza che associa a ogni coppia di punti di $X$ un numero reale non negativo che rappresenta la distanza tra i punti
	\end{itemize}
	Un esempio pu├▓ essere lo \textbf{spazio euclideo} definito come:
	\[
		(\mathbb{R}^n, d) \quad \text{ con $d=$ distanza euclidea}
	\]

\end{definition}
Lo spazio metrico ├¿ un qualunque insieme nel quale viene definita una distanza, anche detta metrica.

\subsubsection{Palla Aperta / Intorno Sferico}
\begin{definition}{Palla aperta / Intorno Sferico}
	S\begin{itemize}
		\item Sia $\vec{x_0} \in \mathbb{R}^n$ fissato
		\item Sia il raggio $r>0$ dove $r \in \mathbb{R}$
	\end{itemize}
	Si definisce \textcolor{red}{palla aperta} o intorno sferico di centro $\vec{x_0}$ e raggio $r$, l'insieme:
	\[
		B(\vec{x_0},r)=\left\{\vec{x} \in \mathbb{R}^n \mid d(\vec{x}, \vec{x_0})<r\right\}
	\]
\end{definition}
Esempi sugli intorni sferici:
\begin{itemize}
	\item In $\mathbb{R}$ gli intorni sferici di $x_0$ non sono altro che intervalli aperti $(x_0-r, x_0+r)$
	\item In $\mathbb{R}^2$ gli intorni sferici di $(x_0,y_0)$ sono dischi (non "cerchi"\footnote{perch├¿ con "cerchi" si intende la circonferenza (il bordo), mentre disco ├¿ l'interno.}) centrati in $(x_0,y_0)$, privi di circonferenza
	\item In $\mathbb{R}^3$ gli intorni sferici di $(x_0,y_0,z_0)$ sono sfere piene \footnote{Sfera piena = interno senza il bordo (come il disco per $\mathbb{R}^2$); Sfera = solo il bordo (come il cerchio per $\mathbb{R}^2$).}
	      centrate in $(x_0,y_0,z_0)$ prive di bordo
\end{itemize}

\subsubsection{Punto interno, esterno, di frontiera, di accumulazione}
\begin{definition}{Punto interno in $\mathbb{R}$}
	UUn \textcolor{red}{punto interno} di un insieme ├¿ un punto per il quale esiste almeno un intorno interamente contenuto nell'insieme. \\ \\
	Sia $x_0 \in A \subseteq \mathbb{R}$ (punto appartenente all'insieme $A$), si dice che $x_0$ ├¿ un punto interno ad $A$ se esiste un intorno di centro $x_0$ e raggio $r$
	completamente contenuto in A:
	\[
		\exists r>0 \mid B(x_0,r) \subset A
	\]
\end{definition}
\begin{definition}{Punto interno in $\mathbb{R}^n$}
	SSia $A \subseteq \mathbb{R}^n$. Un punto $\vec{x_0} \in \mathbb{R}^n$ si dice \textcolor{red}{interno ad A} se $\exists$ un intorno sferico di centro $\vec{x_0}$ e
	raggio\footnote{$r \in \mathbb{R}, r > 0$. Il raggio ├¿ numero reale positivo, non un vettore come $\vec{x_0}$} $r \in \mathbb{R}^+$ completamente contenuto in $A$:
	\[
		\exists r>0 \mid B(\vec{x_0},r) \subset A
	\]
\end{definition}
Un punto interno ad $A$ deve necessariamente appartenere all'insieme.
\begin{definition}{Punto esterno in $\mathbb{R}$}
	UUn \textcolor{red}{punto esterno} ad un insieme ├¿ un punto per il quale esiste almeno un intorno interamente contenuto nel complementare dell'insieme. \\ \\
	Sia $x_0 \in \mathbb{R}$, si dice che $x_0$ ├¿ un punto esterno ad $A \subseteq \mathbb{R}$ se esiste un intorno di centro $x_0$ e raggio $r$
	completamente contenuto nel complementare di A:
	\[
		\exists r>0 \mid B(x_0,r) \subset \overline{A}
	\]
\end{definition}
\begin{definition}{Punto esterno in $\mathbb{R}^n$}
	SSia $A \subseteq \mathbb{R}^n$. Un punto $\vec{x_0} \in \mathbb{R}^n$ si dice \textcolor{red}{esterno ad A} se $\exists$ un intorno sferico di centro $\vec{x_0}$ e
	raggio\footnote{$r \in \mathbb{R}, r > 0$. Il raggio ├¿ numero reale positivo, non un vettore come $\vec{x_0}$} $r \in \mathbb{R}^+$ completamente contenuto in $\overline{A}$\footnote{complementare di $A$}:
	\[
		\exists r>0 \mid B(\vec{x_0},r) \subset \overline{A}
	\]
\end{definition}
Un punto esterno ad $A$ non appartenere all'insieme.
\begin{definition}{Punti di frontiera in $\mathbb{R}$}
	UUn \textcolor{red}{punto di frontiera} (bordo) di un insieme ├¿ un punto il cui ogni intorno contiene almeno un punto dell'insieme e del suo complementare. \\ \\
	Sia $x_0 \in \mathbb{R}$, si dice che $x_0$ ├¿ un punto di frontiera di $A \subseteq \mathbb{R}$ se ogni intorno di centro $x_0$ e raggio $r$
	contiene almeno un punto di $A$ e allo stesso tempo, almeno un punto di $\overline{A}$:
	\[
		\forall{r}>0, \quad \exists y_1 \in A, \quad \exists y_2 \in \overline{A} \quad \text{t.c.} \quad  y_1,y_2 \in B(x_0,r)
	\]
\end{definition}

\begin{definition}{Punti di frontiera in $\mathbb{R}^n$}
	SSia $A \subseteq \mathbb{R}^n$. Un punto $\vec{x_0} \in \mathbb{R}^n$ si dice \textcolor{red}{di frontiera per $A$}
	se ogni intorno sferico di centro $\vec{x_0}$ e
	raggio\footnote{$r \in \mathbb{R}, r > 0$. Il raggio ├¿ numero reale positivo, non un vettore
		come $\vec{x_0}$} $r \in \mathbb{R}^+$ contiene almeno un punto di $A$ e un punto di $\overline{A}$:
	\[
		\forall{r}>0, \quad \exists \vec{y_1} \in A, \quad \exists \vec{y_2} \in \overline{A} \quad \text{t.c.} \quad  \vec{y_1},\vec{y_2} \in B(\vec{x_0},r)
	\]
\end{definition}
Un punto di frontiera per $A$ pu├▓ appartenere o non appartenere all'insieme.

\begin{definition}{Punto di accumulazione}
	UUn punto $\vec{x_0}$ si dice di \textcolor{red}{accumulazione} per $A \subseteq \mathbb{R}^n$ se in ogni intorno di $\vec{x_0}$ esiste un punto di $A$ diverso da $\vec{x_0}$:
	\[
		\forall{r} >0 \quad B(\vec{x_0},r) \cap (A \setminus \left\{\vec{x_0}\right\}) \neq \emptyset
	\]
\end{definition}

\subsubsection{Insieme limitato, aperto, chiuso}
\begin{definition}{Insieme limitato}
	DDefinisco l'insieme $A \subseteq \mathbb{R}^n$ \textcolor{red}{limitato} se $\exists$ una palla aperta di centro $\vec{x_0} \in \mathbb{R}^n$ e raggio $r>0$ in cui $A$ ne risulta interamente contenuto:
	\[
		A \subset B(\vec{x_0},r)
	\]
\end{definition}

\begin{definition}{Insieme aperto}
	UUn insieme $A \subseteq \mathbb{R}$ si dice \textcolor{red}{aperto} se ogni $x \in A$ ├¿ un punto interno\footnote{per ogni punto appartenente ad $A$ esiste un intorno completamente
		contenuto nell'insieme (ne basta trovare uno per ogni punto).}:
	\[
		\forall{x} \in A, \quad \exists r>0 \mid B(x_0, r) \subset A
	\]
\end{definition}

\begin{definition}{Insieme chiuso}
	UUn insieme $A \subseteq \mathbb{R}$ si dice \textcolor{red}{chiuso} se $\overline{A}$ ├¿ aperto:
	\[
		A \text{ chiuso se } \quad \forall{x} \in \overline{A}, \quad \exists r>0 \mid B(x_0, r) \subset \overline{A}
	\]
\end{definition}

Esempi di insiemi aperti e chiusi:

\begin{itemize}
	\item \textbf{In $\mathbb{R}$:}
	      \begin{itemize}
		      \item Aperto: $(0,1)$
		      \item Chiuso: $[0,1]$
	      \end{itemize}

	\item \textbf{In $\mathbb{R}^2$:}
	      \begin{itemize}
		      \item Aperti:
		            \[\left\{(x,y) \mid x^2+y^2 < 1 \right\}, \quad
			            \left\{(x,y) \mid x^2+y^2 > 1 \right\}, \quad
			            \left\{(x,y) \mid x \neq 1 \land y \neq 0 \right\}\]
		      \item Chiusi:
		            \[\left\{(x,y) \mid x^2+y^2 \leq 1 \right\}, \quad
			            \left\{(x,y) \mid x^2+y^2 \geq 1 \right\}, \quad
			            \left\{(x,y) \mid x^2+y^2 = 1 \right\}\]
		      \item N├® aperto n├® chiuso:
		            \[\left\{(x,y) \mid x^2+y^2 \leq 1 \land x > 0 \right\}\]
	      \end{itemize}
\end{itemize}

\textbf{Complementare:} se un insieme ├¿ aperto, il suo complementare ├¿ chiuso e viceversa:
\[\mathbb{R}^n \text{ aperto} \implies \emptyset \text{ chiuso}, \qquad \mathbb{R}^n \text{ chiuso} \implies \emptyset \text{ aperto}\]

\begin{theorem}{Criterio del confronto}
	HHo trovato il limite:
	\[
		\lim_{\rho \to 0^+} f(x_0 + \rho \cos \theta, y_0 + \rho \sin \theta) = \ell
	\]
	per dimostrarlo, ├¿ sufficiente trovare una $g(\rho) \geq 0$ tale che valga quanto segue:
	\[
		0 \leq |f(x_0 + \rho \cos \theta, y_0 + \rho \sin \theta) - \ell | \leq g(\rho)
	\]
	dove:
	\[
		\lim_{\rho \to 0^+} g(\rho) = 0
	\]
\end{theorem}
\begin{esempio}[Esempio: come usare il criterio del confronto]
	\[
		\lim_{(x,y) \to (0,0)} \frac{x^2y}{x^2+y^2}
	\]
	Trasformo in coordinate polari e trovo il limite:
	\begin{align*}
		\lim_{\rho \to 0^+} \frac{(\rho \cos \theta)^2 \rho \sin \theta}{(\rho \cos \theta)^2+(\rho \sin \theta)^2}
		 & = \lim_{\rho \to 0^+} \frac{\rho^3 \cos^2\theta \sin\theta}{\rho^2(\cos^2\theta + \sin^2\theta)} \\
		 & = \lim_{\rho \to 0^+} \frac{\rho^3 \cos^2\theta \sin\theta}{\rho^2}                              \\
		 & = \lim_{\rho \to 0^+} \rho \cos^2\theta \sin\theta                                               \\
		 & = 0
	\end{align*}
	\textbf{Dimostro che il limite esiste e fa zero:} \\
	\textcolor{red}{Trovo $g(\rho)$} \\ \\
	\textcolor{cyan}{Per prima cosa imposto valore assoluto:}
	\[
		|\cos ^2 \theta \sin \theta|
	\]
	\textcolor{cyan}{e sapendo che $\forall x$ vale che $0 \leq |x|$, allora:}
	\[
		0 \leq |\cos ^2 \theta \sin \theta|
	\]
	\textcolor{cyan}{per la propriet├á del valore assoluto\footnote{$|x \cdot y| = |x| \cdot |y| \quad \forall x,y \in \mathbb{R}$} ho quindi che:}
	\[
		0 \leq |\cos ^2 \theta \sin \theta| = |\cos^2 \theta| \cdot |\sin \theta|
	\]
	\textcolor{cyan}{vale sempre che\footnote{$-1 \leq |\cos\theta|\leq 1$ e $-1 \leq |\sin\theta|\leq 1$} $|\cos\theta|\leq 1$ e $|\sin\theta|\leq 1 \quad \forall \theta$,  quindi:}
	\[
		0 \leq |\cos ^2 \theta \sin \theta| = |\cos^2 \theta| \cdot |\sin \theta| \leq 1 \cdot 1
	\]
	\textcolor{cyan}{e quindi:}
	\[
		0 \leq |\cos ^2 \theta \sin \theta| = |\cos^2 \theta| \cdot |\sin \theta| \leq 1
	\]
	\textcolor{cyan}{quindi:}
	\[
		0 \leq |\cos ^2 \theta \sin \theta|  \leq 1
	\]
	\textcolor{cyan}{introduco $\rho$:}
	\[
		0 \leq |\textcolor{red}{\rho} \cos ^2 \theta \sin \theta|  \leq \textcolor{red}{\rho} \cdot 1
	\]
	\textcolor{cyan}{ho trovato $g(\rho)$:}
	\[
		0 \leq |\rho \cos ^2 \theta \sin \theta|  \leq \underbrace{\rho}_{g(\rho)}
	\]
	e sapendo che la $g(\rho)=\rho$ trovata dipende solo da $\rho$ e che $\lim_{\rho \to 0^+} g(\rho) = 0$, dimostro che il limite
	esiste ed ├¿ quello trovato (zero):
	\[
		0 \leq |\rho \cos ^2 \theta \sin \theta - \ell |  \leq \rho \cdot 1
	\]
	\[
		0 \leq |\rho \cos ^2 \theta \sin \theta - 0 |  \leq \rho
	\]
\end{esempio}

\subsubsection{Continuit├á per funzioni in pi├╣ variabili}
\begin{definition}{Continuit├á in $\mathbb{R}^2$}
	SSia $f: A \subseteq \mathbb{R}^2 \to \mathbb{R}$ e sia $P_0=(x_0,y_0) \in A$ e punto di accumulazione per $A$. \\
	Si dice che la funzione \textcolor{red}{$f$ ├¿ continua in $P_0$} se:
	\[
		\lim_{(x,y) \to (x_0,y_0)} f(x,y) = f(x_0,y_0)
	\]
\end{definition}
La funzione ├¿ continua in un punto $P_0$ (tale punto deve appartenere all'insieme $\land$ deve essere un punto di accumulazione per l'insieme) quando il limite della funzione per $(x,y)$ tendente a $P_0$ coincide col valore di $f$ valutata nel punto $P_0$.

\subsection{Curva in Analisi Matematica}
Una curva ├¿ un funzione continua su un intervallo cosi definita:
\[
	\gamma: [a,b] \subseteq \mathbb{R} \to \mathbb{R}^n
\]
La parametrizzazione ├¿  un modo di descrivere la curva attraverso un gruppo di funzioni\footnote{
	\begin{itemize}
		\item $x(t),y(t)$ in $\mathbb{R}^2$
		\item $x(t),y(t),z(t)$ in $\mathbb{R}^3$
		\item $x_1(t),x_2(t), .., x_n(t)$ in $\mathbb{R}^n$
	\end{itemize}}  che variano in base al parametro $t$:
\[
	\textcolor{red}{\gamma(t)}= \left(x_1(t),x_2(t), \cdots, x_n(t) \right) =
	\begin{cases}
		x_1 = x_1(t) \\
		x_2 = x_2(t) \\
		\vdots       \\
		x_n= x_n(t)
	\end{cases}
\]
\begin{definition}{Sostegno di una curva}
	LL'immagine di una curva ├¿ chiamata \textcolor{red}{sostegno} della curva.
\end{definition}
\begin{osservazioni}[Curva vs Sostegno]
	Per "curva" non si intende il sostegno (e quindi la sua immagine) ma la funzione.
	├ê importante non confondere le due definizioni. Per esempio, prendiamo la circonferenza:
	\begin{center}
		\includegraphics[width=0.6\linewidth]{curvavssostegno.png}
	\end{center}

\end{osservazioni}

Posso avere:
\begin{itemize}
	\item Curvatura $=0$: come una retta
	\item Curvatura $\neq 0$ come una semicirconferenza, un semiellisse, ecc..
\end{itemize}
\begin{definition}{Equazioni Parametriche $\mathbb{R}^2$}
	SSia $\gamma$ la curva dipendente dal parametro $t$ e la sua rappresentazione parametrica:
	\[
		\gamma: [a,b] \subseteq \mathbb{R} \to \mathbb{R}^2
	\]
	\[
		\textcolor{red}{\gamma(t)}= \left(x(t),y(t)\right) =
		\begin{cases}
			x = x(t) \\
			y = y(t)
		\end{cases}
	\]
\end{definition}
\begin{definition}{Equazioni Parametriche $\mathbb{R}^3$}

\subsection{Curva Regolare}
Una curva si dice \textcolor{red}{regolare} quando la norma del vettore derivato non si annulla per nessun valore di $t$.
\begin{esempio}[Esempio di curva regolare]
	Sia la curva $\gamma: [0, \pi] \to \mathbb{R}^3$ e la sua rappresentazione parametrica:
	\[
		\gamma(t) = \begin{cases}
			x = \sin(t) \\
			y = \cos(t) \\
			z = e^t
		\end{cases} \quad t \in [0, \pi]
	\]
	\begin{enumerate}
		\item Calcolo il vettore derivato:
		      \[
			      \gamma'(t) = \begin{cases}
				      x = \cos(t)  \\
				      y = -\sin(t) \\
				      z = e^t
			      \end{cases} \quad t \in [0, \pi]
		      \]
		\item Calcolo la norma:
		      \begin{align*}
			      \|\gamma'(t)\|
			       & = \sqrt{(\cos(t))^2+(-\sin(t))^2+(e^t)^2} \\
			       & = \sqrt{\cos^2(t)+\sin^2(t)+e^{2t}}       \\
			       & = \sqrt{1+e^{2t}}
		      \end{align*}
		\item Impongo uguale a zero:
		      \begin{align*}
			      \sqrt{1+e^{2t}}                & = 0   \\
			      \left(\sqrt{1+e^{2t}}\right)^2 & = 0^2 \\
			      1+e^{2t}                       & = 0   \\
			      e^{2t}                         & = -1  \\
		      \end{align*}
		      \text{nessun valore di } t \text{ soddisfa l'equazione, quindi la curva ├¿ regolare.} \textcolor{green}{\checkmark}
	\end{enumerate}
\end{esempio}

\section{Calcolo differenziale per funzioni in pi├╣ variabili}
\begin{osservazioni}[Derivabilit├á vs Differenziabilit├á]
	\begin{itemize}
		\item Per funzioni da $\mathbb{R} \to \mathbb{R}$, derivabilit├á $=$ differenziabilit├á
		\item Per funzioni da $\mathbb{R}^n \to \mathbb{R}$,  derivabilit├á $\neq$ differenziabilit├á
	\end{itemize}
\end{osservazioni}
\begin{osservazioni}
	\begin{itemize}
		\item Per $\mathbb{R} \to \mathbb{R}$, derivabile $\implies$ continua (non il contrario)
		\item Per $\mathbb{R}^n \to \mathbb{R}$, derivabile $\centernot{\implies}$ continua, differenziabile $\implies$ continua $\land$ derivabile
	\end{itemize}
\end{osservazioni}
Il concetto di differenziabilit├á ├¿ pi├╣ forte in $\mathbb{R}^n$ perch├¿ implica che, se la funzione ├¿ differenziabile in un punto, allora ├¿ anche continua e derivabile in quel punto.
Mentre in $\mathbb{R}$, la derivabilit├á in un punto implica solo la continuit├á. \\
Si fa una distinzione tra derivabilit├á e differenziabilit├á perch├¿ quando si hanno funzioni con pi├╣ variabili non esiste \textcolor{red}{\textbf{la}} derivata ma \textcolor{red}{\textbf{le}} derivate direzionali.

\subsection{Teoremi Fondamentali del Calcolo Differenziale in $\mathbb{R}$}
\subsubsection{Teorema di Fermat}
Questo teorema stabilisce che se una funzione ammette un massimo o un minimo locale in un punto interno del dominio, e se derivabile in quel punto, allora la derivata prima
valutata in tal punto ├¿ nulla. \\
\begin{theorem}{Teorema di Fermat}
	SSia $f:\left(a,b\right) \to \mathbb{R}$ e sia $x_0 \in \left(a,b\right)$. \\
	Se nel punto $x_0$ la funzione ├¿ derivabile $\land \; x_0$ ├¿ un punto di minimo o massimo locale, allora:
	\[
		f'(x_0)=0
	\]
\end{theorem}
\subsubsection{Teorema di Weierstrass}
Il teorema indica che ogni funzione reale continua in un intervallo chiuso e limitato ammette (almeno) un massimo $\land$ un minimo assoluto. \\
\begin{theorem}{Teorema di Weierstrass}
	SSia $f:[a,b] \to \mathbb{R}$ una funzione continua, allora $\exists \; x_M,x_m \in [a,b]$ tali che:
	\[
		f(x_M)=M=\underset{[a,b]}{\max } f, \quad \quad \quad f(x_m)=m=\underset{[a,b]}{\min } f
	\]
\end{theorem}
\subsubsection{Teorema di Rolle}
Il teorema afferma che quando una funzione ├¿ continua e derivabile in un intervallo limitato, e tale funzione assume lo stesso valore nei due estremi di tale intervallo,
allora esiste almeno un punto interno all'intervallo dove il valore della derivata si annulla.
\begin{theorem}{Teorema di Rolle}
	SSia $f:[a,b] \to \mathbb{R}$ che soddisfa le seguenti condizioni:
	\begin{enumerate}
		\item $f$ continua in $[a,b]$
		\item $f$ derivabile in $(a,b)$
		\item $f(a)=f(b)$ (assume lo stesso valore agli estremi dell'intervallo)
	\end{enumerate}
	allora $\exists$ un punto $c \in (a,b)$ tale che:
	\[
		f'(c)=0
	\]
\end{theorem}
\subsection{Derivate Direzionali e Parziali}
\begin{osservazioni}[Ripasso: definizione derivata in $\mathbb{R}$]
	La derivata di una funzione in un punto del suo dominio ├¿ il limite del rapporto incrementale di $h \to 0$:
	\[
		f'(x_0)=\frac{df}{dx}(x_0)=\lim_{h \to 0}\frac{f(x_0+h)-f(x_0)}{h}
	\]
\end{osservazioni}
\subsubsection{Derivata Direzionale}
Il concetto di derivata direzionale si riferisce al concetto di derivazione in due variabili. Non esiste un'unica derivata in due variabili, ma ne esiste una per ogni direzione.
\begin{definition}{Derivata Direzionale in $\mathbb{R}^2$}
	SSiano: \begin{itemize}
		\item $f: \mathbb{R}^2 \supseteq A \to \mathbb{R}$ con $A$ aperto
		\item $(x,y) \in A$
		\item $\vec{v} = (v_1, v_2)$ versore (norma unitaria)
	\end{itemize}
	Si definisce $\frac{\partial f}{\partial \vec{v}}(x,y)$ \textcolor{red}{derivata direzionale} di $f(x,y)$ lungo la direzione $\vec{v}$ il limite, se esiste finito:
	\[
		\frac{\partial f}{\partial \vec{v}}(x,y)=\lim_{h \to 0}\frac{f(x+hv_1,y+hv_2)-f(x,y)}{h}
	\]
	\\ Se il limite esiste ed ├¿ finito, $f$ si dice derivabile nella direzione $\vec{v}$ in $(x,y)$
\end{definition}
\begin{definition}{Derivata Direzionale in $\mathbb{R}^n$}
	SSiano: \begin{itemize}
		\item $f: \mathbb{R}^n \supseteq A \to \mathbb{R}$ con $A$ aperto
		\item $\vec{x}=(x_1,x_2,...,x_n) \in A$
		\item $\vec{v} = (v_1, v_2,...,v_n)$ versore in $\mathbb{R}^n$
	\end{itemize}
	Si definisce $\frac{\partial f}{\partial \vec{v}}(\vec{x})$ \textcolor{red}{derivata direzionale} di $f(\vec{x})$ lungo la direzione $\vec{v}$ il limite, se esiste finito:
	\[
		\frac{\partial f}{\partial \vec{v}}(\vec{x})=\lim_{h \to 0}\frac{f(\vec{x}+h\vec{v})-f(\vec{x})}{h}=\lim_{h \to 0}\frac{f(x_1+hv_1,...,x_n+hv_n)-f(x_1,...,x_n)}{h}
	\]
	\\
	Se il limite esiste ed ├¿ finito, $f$ si dice derivabile nella direzione $\vec{v}$ in $\vec{x}$
\end{definition}
\begin{esercizio}[Esercizio]
	Sia $f(x,y)=e^x y$ lungo il vettore $\vec{u}=(3,4)$. Calcola derivata direzionale nel punto $(2,0)$.
	\begin{enumerate}
		\item \textbf{Controllo che il vettore abbia norma unitaria:}
		      \begin{align*}
			      \|u\|          & \overset{?}=1 \\
			      \sqrt{3^2+4^2} & =1            \\
			      5              & \neq 1
		      \end{align*}
		      Il vettore non ha norma unitaria, quindi devo normalizzarlo:
		      \begin{align*}
			      \vec{v}=\frac{\vec{u}}{\|u\|}=\left(\frac{u_1}{\|u\|},\frac{u_2}{\|u\|}\right)=\left(\frac{3}{5},\frac{4}{5}\right)
		      \end{align*}
		\item \textbf{Calcolo il limite:}
		      \begin{align*}
			      \frac{\partial f}{\partial \vec{v}}(2,0) & = \lim_{h \to 0} \frac{f(x+hv_1,y+hv_2)-f(x,y)}{h}                              \\
			                                               & = \lim_{h \to 0} \frac{f(2+h\frac{3}{5},0+h\frac{4}{5})-f(2,0)}{h}              \\
			                                               & = \lim_{h \to 0} \frac{f(2+h\frac{3}{5},h\frac{4}{5})-(e^2 \cdot 0)}{h}         \\
			                                               & = \lim_{h \to 0} \frac{f(2+h\frac{3}{5},h\frac{4}{5})}{h}                       \\
			                                               & = \lim_{h \to 0} \frac{e^{2+h\frac{3}{5}} \cdot  \left(h \frac{4}{5}\right)}{h} \\
			                                               & = \frac{4}{5}e^2
		      \end{align*}
	\end{enumerate}
	Pertanto la derivata direzionale ├¿:
	\[
		\frac{\partial f}{\partial \vec{v}}(2,0)=\frac{4}{5}e^2
	\]
\end{esercizio}

\subsubsection{Derivata Parziale}
La derivata parziale ├¿ un caso speciale di derivata direzionale.
\begin{definition}{Derivata Parziale in $\mathbb{R}^2$}
	SSiano: \begin{itemize}
		\item $f: \mathbb{R}^2 \supseteq A \to \mathbb{R}$ con $A$ aperto
		\item $(x,y) \in A$
	\end{itemize}
	Si definiscono $\frac{\partial f}{\partial x}(x,y)$ \textcolor{red}{derivata parziale rispetto a $x$ nel punto $(x,y)$} e
	$\frac{\partial f}{\partial y}(x,y)$ \textcolor{red}{derivata parziale rispetto a $y$ nel punto $(x,y)$} i limiti, se esistono finiti:
	\[
		\frac{\partial f}{\partial x}(x,y)=\lim_{h \to 0}\frac{f(x+h,y)-f(x,y)}{h}
	\]
	\[
		\frac{\partial f}{\partial y}(x,y)=\lim_{h \to 0}\frac{f(x,y+h)-f(x,y)}{h}
	\]
\end{definition}

\begin{definition}{Derivata Parziale in $\mathbb{R}^n$}
	SSiano: \begin{itemize}
		\item $f: \mathbb{R}^n \supseteq A \to \mathbb{R}$ con $A$ aperto
		\item $\vec{x}=(x_1,x_2,...,x_n) \in A$
	\end{itemize}
	Si definiscono $\frac{\partial f}{\partial x_i}(\vec{x})$ \textcolor{red}{derivate parziale rispetto a $x_i$ nel punto $\vec{x}$} i limiti, se esistono finiti:
	\[
		\frac{\partial f}{\partial x_i}(\vec{x})=\lim_{h \to 0}\frac{f((x_1,\dots,x_i)+h, (x_{i+1},\dots,x_n))-f(x_1,\dots,x_n)}{h} \quad \quad \text{per } i=1,...,n
	\]
\end{definition}
\subsubsection{Piano tangente}
In funzioni di una variabile, definire la derivata di una funzione equivale a definire la retta tangente al suo grafico. Per una funzione di due variabili,
il problema analogo ├¿ definire il \textcolor{red}{piano tangente} al grafico della funzione.  \\ Le due derivate parziali danno due rette tangenti (lungo $x$ e lungo $y$),
e queste determinano un piano, ma quel piano ├¿ tangente al grafico solo se $f$ ├¿ differenziabile in $(x_0,y_0)$.
Se $f$ non ├¿ differenziabile, il piano esiste come oggetto geometrico ma non ├¿ tangente al grafico!
\begin{center}
	Equazione del piano tangente:
\end{center}
\[
	\textcolor{red}{z=\,}f(x_0,y_0)+\frac{\partial f}{\partial x}(x_0,y_0)(x-x_0)+\frac{\partial f}{\partial y}(x_0,y_0)(y-y_0)
\]

\subsubsection{Vettore Gradiente}
\begin{definition}{Vettore Gradiente in $\mathbb{R}^2$}
	SSe $f: \mathbb{R}^2 \supseteq A \to \mathbb{R}$ (con $A$ aperto) ammette le due derivate parziali (derivata parziale rispetto a $x$ e la derivata parziale rispetto a $y$)
	in un punto $(x_0,y_0) \in A$, definisco \textcolor{red}{vettore gradiente} di $f$ in
	$(x_0,y_0)$ il vettore le cui componenti sono le 2 derivate parziali di $f$ in $(x_0,y_0)$:
	\[
		\nabla f(x_0,y_0)=\left(\frac{\partial f}{\partial x}(x_0,y_0), \frac{\partial f}{\partial y}(x_0,y_0) \right) \in \mathbb{R}^2
	\]
\end{definition}
\begin{definition}{Vettore Gradiente in $\mathbb{R}^n$}
	SSe $f: \mathbb{R}^n \supseteq A \to \mathbb{R}$ (con $A$ aperto) ammette $n$ derivate parziali in un punto $\vec{x} \in A$, definisco \textcolor{red}{vettore gradiente} di $f$ in
	$\vec{x}$ il vettore le cui componenti sono le $n$ derivate parziali di $f$ in $\vec{x}$:
	\[
		\nabla f(\vec{x})=\left(\frac{\partial f}{\partial x_1}(\vec{x}),\frac{\partial f}{\partial x_2}(\vec{x}), \dots,\frac{\partial f}{\partial x_n}(\vec{x}) \right) \in \mathbb{R}^n
	\]
\end{definition}
\begin{theorem}{Formula del Gradiente in $\mathbb{R}^2$}
	SSia $f: \mathbb{R}^2 \supseteq A \to \mathbb{R}$ (con $A$ aperto) differenziabile in $(x_0,y_0) \in A$, allora per ogni versore $\vec{v}=(v_1,v_2)$ esiste
	la derivata direzionale in $(x_0,y_0)$ di $f$ lungo la direzione $\vec{v}$ ed ├¿ data dal prodotto scalare tra il vettore gradiente in $(x_0,y_0)$ e il versore $\vec{v}$:
	\[
		\frac{\partial f}{\partial \vec{v}}(x_0,y_0) = \langle \nabla f(x_0,y_0), \vec{v} \rangle = \left(\frac{\partial f}{\partial x}(x_0,y_0) \cdot v_1 + \frac{\partial f}{\partial y}(x_0,y_0) \cdot v_2 \right)
	\]
\end{theorem}
\begin{theorem}{Formula del Gradiente in $\mathbb{R}^n$}
	SSia $f: \mathbb{R}^n \supseteq A \to \mathbb{R}$ (con $A$ aperto) differenziabile in $\vec{x_0} \in A$, allora per ogni versore $\vec{v}$ esiste
	la derivata direzionale in $\vec{x_0}$ di $f$ lungo la direzione $\vec{v}$ ed ├¿ data dal prodotto scalare tra il vettore gradiente in $\vec{x_0}$ e il versore $\vec{v}$:
	\[
		\frac{\partial f}{\partial \vec{v}}(\vec{x_0}) = \langle \nabla f(\vec{x_0}), \vec{v} \rangle = \sum_{i=1}^{n}\left(\frac{\partial f}{\partial x_i}(\vec{x_0}) \cdot v_i \right)
	\]
\end{theorem}

\begin{theorem}{Differenziabilit├á in $\mathbb{R}^2$}
	SSiano:
	\begin{itemize}
		\item $f: A \subseteq \mathbb{R}^2 \to \mathbb{R}$ con $A$ aperto
		\item $(x_0,y_0) \in A$
		\item $(h,k) \in \mathbb{R}^2$ il vettore incremento
	\end{itemize}
	Si dice che $f$ ├¿ \textcolor{red}{differenziabile in $(x_0,y_0)$} se:
	\[
		\lim_{(h,k) \to (0,0)} \frac{f(x_0+h, y_0+k)-f(x_0,y_0)- \frac{\partial f}{\partial x}(x_0,y_0)h - \frac{\partial f}{\partial y}(x_0,y_0)k}{\sqrt{h^2+k^2}} = 0
	\]
	o meglio in notazione compatta:
	\[
		\lim_{(h,k) \to (0,0)} \frac{f(x_0+h, y_0+k)-f(x_0,y_0)- \langle \nabla f(x_0,y_0), (h,k) \rangle}{\|(h,k)\|} = 0
	\]
	\\
	Quindi se il limite di $(h,k) \to (0,0)$ del rapporto tra:
	\begin{itemize}
		\item numeratore: funzione valutata nel punto pi├╣ uno spostamento $f(x_0+h, y_0+k)$ - funzione valutata nel punto $(x_0,y_0)$ - il prodotto scalare tra il gradiente della funzione in $(x_0,y_0)$ e il vettore $(h,k)$
		\item denominatore: norma del vettore $(h,k)$
	\end{itemize}
	├¿ uguale a zero $\implies f$ ├¿ differenziabile in $(x_0,y_0)$.
\end{theorem}

\begin{definition}{Differenziale}
	SSe $f$ ├¿ differenziabile in $(x_0,y_0)$, l'applicazione lineare  $df: \mathbb{R}^2 \to \mathbb{R}$ definita da:
	\[
		df(x_0,y_0)(h,k) = \langle \nabla f(x_0,y_0), (h,k) \rangle
	\]
	├¿ chiamata \textcolor{red}{differenziale di $f$ in $(x_0,y_0)$}.
\end{definition}

\begin{theorem}{Differenziabilit├á in $\mathbb{R}^n$}
	SSiano
	\begin{itemize}
		\item $f: A \subseteq \mathbb{R}^n \to \mathbb{R}$ con $A$ aperto
		\item $\vec{x}=(x_1,\dots,x_n) \in A$
		\item $\vec{h}=(h_1,\dots,h_n) \in \mathbb{R}^n$ il vettore incremento
	\end{itemize}
	Si dice che $f$ ├¿ \textcolor{red}{differenziabile in $\vec{x}$} se:
	\[
		\lim_{\vec{h} \to \vec{0}} \frac{f(\vec{x}+\vec{h})-f(\vec{x})- \left(\sum_{i=1}^{n}\frac{\partial f}{\partial x_i}(\vec{x})\cdot h_i\right)}{\sqrt{\left(\sum_{i=1}^{n}h_i ^2\right)}} = 0
	\]
	in notazione compatta:
	\[
		\lim_{\vec{h} \to \vec{0}} \frac{f(\vec{x}+\vec{h})-f(\vec{x})- \langle \nabla f(\vec{x}), \vec{h} \rangle}{\|\vec{h}\|} = 0
	\]
\end{theorem}
\begin{definition}{Differenziale}
	SSe $f$ ├¿ differenziabile in $(\vec{x})$, l'applicazione lineare  $df: \mathbb{R}^n \to \mathbb{R}$ definita da:
	\[
		df(\vec{x})(\vec{h}) = \langle \nabla f(\vec{x}), \vec{h} \rangle
	\]
	├¿ chiamata \textcolor{red}{differenziale di $f$ in $(\vec{x})$}.
\end{definition}

\begin{osservazioni}[Derivate Parziali Seconde]
	Siano le derivate prime parziali:
	\[
	\frac{\partial f}{\partial x}(x,y) \quad \quad \frac{\partial f}{\partial y}(x,y)
	\]
	Le derivate parziali seconde si calcolano derivando le derivate prime parziali.
	\begin{itemize}
		\item \textcolor{red}{$f_{xx}(x,y) = \,$} $\frac{\partial}{\partial x}(x,y)\left(\frac{\partial f}{\partial x}(x,y)\right) = \frac{\partial^2 f}{\partial x^2}(x,y)$ 
		\item \textcolor{red}{$f_{yx}(x,y) = \,$} $\frac{\partial}{\partial x}(x,y)\left(\frac{\partial f}{\partial y}(x,y)\right) = \frac{\partial^2 f}{\partial x \partial y}(x,y)$ 
		\item \textcolor{red}{$f_{xy}(x,y) = \,$} $\frac{\partial}{\partial y}(x,y)\left(\frac{\partial f}{\partial x}(x,y)\right) = \frac{\partial^2 f}{\partial y \partial x}(x,y)$ 
		\item \textcolor{red}{$f_{yy}(x,y) = \,$} $\frac{\partial}{\partial y}(x,y)\left(\frac{\partial f}{\partial y}(x,y)\right) = \frac{\partial^2 f}{\partial y^2}(x,y)$
	\end{itemize}
	Generalizzando:
	\[
	f_{x_ix_j} = \frac{\partial}{\partial x_j}\left(\frac{\partial f}{\partial x_i}\right)
	\]
\end{osservazioni}

\subsubsection{Teorema di Weierstrass}
\begin{osservazioni}[Teorema di Weierstrass in $\mathbb{R}$]
		Sia $f:[a,b] \to \mathbb{R}$ una funzione continua, allora $\exists \; x_M,x_m \in [a,b]$ tali che:
	\[
		f(x_M)=M=\underset{[a,b]}{\max } f, \quad \quad \quad f(x_m)=m=\underset{[a,b]}{\min } f
	\]
\end{osservazioni}

\begin{theorem}{Teorema di Weierstrass in $\mathbb{R}^n$}
	SSia $f: W \subset \mathbb{R}^n \to \mathbb{R}$ continua con $W$ insieme chiuso e limitato, allora $\exists \, \vec{x}_M,\vec{x}_m \in W$ tali che:
	\[
	f(\vec{x}_m) \leq f(\vec{x}) \leq f(\vec{x}_M) \quad \quad \forall{\vec{x}} \in W
	\] 
\end{theorem}
